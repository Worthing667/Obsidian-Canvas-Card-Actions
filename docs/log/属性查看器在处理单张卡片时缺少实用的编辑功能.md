## 问题分析

当前的 `CardPropertiesModal` 设计主要面向多卡片场景，对于单卡片情况存在以下限制：

1. **批量操作区域缺失**：代码中的 `createBatchActions` 方法仅在卡片数量大于1时执行
2. **应用按钮无效**：`applyCardSize` 方法会排除源卡片本身，导致单卡片时无目标可应用
3. **缺少直接编辑功能**：无法直接修改当前卡片的尺寸

## 解决方案

需要修改 `CardPropertiesModal` 类，为单卡片场景添加专门的编辑功能。以下是具体的改进方案：

// 在 CardPropertiesModal.ts 中修改 createBatchActions 方法

createBatchActions(container) {
  if (this.cardInfos.length === 1) {
    // 单卡片编辑模式

createSingleCardEditor(container) {
  const actionsTitle = container.createEl("h3", { text: "编辑卡片尺寸" });
  
  const currentCard = this.cardInfos[0];
  
  // 当前尺寸显示
  const currentSizeDiv = container.createDiv({ cls: "current-size-display" });
  currentSizeDiv.innerHTML = `
    <div class="size-info">
      <span class="size-label">当前尺寸：</span>
      <span class="size-value">${currentCard.width} × ${currentCard.height} px</span>
    </div>
  `;
  
  // 快速尺寸选择
  const quickSizeDiv = container.createDiv({ cls: "quick-size-section" });
  quickSizeDiv.createEl("h4", { text: "常用尺寸" });
  
  const quickSizeButtons = quickSizeDiv.createDiv({ cls: "quick-size-buttons" });
  
  const commonSizes = [
    { name: "小卡片", width: 250, height: 200 },
    { name: "标准卡片", width: 400, height: 300 },
    { name: "大卡片", width: 500, height: 400 },
    { name: "宽卡片", width: 600, height: 200 },
    { name: "高卡片", width: 300, height: 500 }
  ];
  
  commonSizes.forEach(size => {
    const button = quickSizeButtons.createEl("button", { 
      text: `${size.name}\n${size.width}×${size.height}`,
      cls: "quick-size-btn"
    });
    
    // 如果是当前尺寸，添加高亮
    if (size.width === currentCard.width && size.height === currentCard.height) {
      button.addClass("current-size");
    }
    
    button.addEventListener("click", async () => {
      await this.applySingleCardSize(size.width, size.height);
    });
  });
  
  // 自定义尺寸编辑
  const customSizeDiv = container.createDiv({ cls: "custom-size-editor" });
  customSizeDiv.createEl("h4", { text: "自定义尺寸" });
  
  const inputGroup = customSizeDiv.createDiv({ cls: "input-group" });
  
  const widthInput = inputGroup.createEl("input", {
    type: "number",
    value: currentCard.width.toString(),
    attr: { min: "50", max: "2000", step: "10" }
  });
  widthInput.style.width = "80px";
  
  inputGroup.createSpan({ text: " × " });
  
  const heightInput = inputGroup.createEl("input", {
    type: "number", 
    value: currentCard.height.toString(),
    attr: { min: "50", max: "2000", step: "10" }
  });
  heightInput.style.width = "80px";
  
  const applyBtn = inputGroup.createEl("button", {
    text: "应用",
    cls: "mod-cta"
  });
  
  const resetBtn = inputGroup.createEl("button", {
    text: "重置"
  });
  
  // 事件处理
  applyBtn.addEventListener("click", async () => {
    const width = parseInt(widthInput.value);
    const height = parseInt(heightInput.value);
    
    if (this.validateSize(width, height)) {
      await this.applySingleCardSize(width, height);
    }
  });
  
  resetBtn.addEventListener("click", () => {
    widthInput.value = currentCard.width.toString();
    heightInput.value = currentCard.height.toString();
  });
  
  // 添加键盘支持
  [widthInput, heightInput].forEach(input => {
    input.addEventListener("keypress", async (e) => {
      if (e.key === "Enter") {
        applyBtn.click();
      }
    });
    
    input.addEventListener("input", () => {
      const width = parseInt(widthInput.value);
      const height = parseInt(heightInput.value);
      applyBtn.disabled = !this.validateSize(width, height);
    });
  });
}

// 新增方法：应用单卡片尺寸
async applySingleCardSize(width, height) {
  try {
    if (!this.validateSize(width, height)) {
      new Notice("尺寸值无效，请输入50-2000之间的数值");
      return;
    }
    
    const targetCard = this.cards[0]; // 单卡片情况下只有一张卡片
    await this.cardService.unifyCardSizes([targetCard], { width, height });
    
    new Notice(`卡片尺寸已更新为 ${width}×${height}`);
    this.close();
    
  } catch (error) {
    console.error("更新单卡片尺寸失败:", error);
    new Notice("更新尺寸失败: " + error.message);
  }
}

// 新增方法：尺寸验证
validateSize(width, height) {
  return width >= 50 && width <= 2000 && 
         height >= 50 && height <= 2000 && 
         !isNaN(width) && !isNaN(height);
}
