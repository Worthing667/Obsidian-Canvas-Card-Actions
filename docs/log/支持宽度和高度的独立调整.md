      <span class="size-label">当前尺寸：</span>

  });

    text: "重置",

    applyBothBtn.disabled = !(this.validateDimension(width) && this.validateDimension(height));
  });

// 新增方法：应用单个维度的尺寸
async applySingleDimension(dimension, value) {
  try {
    const currentCard = this.cardInfos[0];
    const targetCard = this.cards[0];
    
    let newWidth, newHeight;
    
    if (dimension === "width") {
      newWidth = value;
      newHeight = currentCard.height; // 保持原有高度
    } else {
      newWidth = currentCard.width; // 保持原有宽度
      newHeight = value;
    }
    
    await this.cardService.unifyCardSizes([targetCard], { width: newWidth, height: newHeight });
    
    new Notice(`卡片${dimension === "width" ? "宽度" : "高度"}已更新为 ${value}px`);
    
    // 更新当前显示的信息
    this.updateCurrentCardInfo(newWidth, newHeight);
    
  } catch (error) {
    console.error("更新卡片尺寸失败:", error);
    new Notice("更新尺寸失败: " + error.message);
  }
}

// 更新当前卡片信息显示
updateCurrentCardInfo(newWidth, newHeight) {
  this.cardInfos[0].width = newWidth;
  this.cardInfos[0].height = newHeight;
  
  // 更新当前尺寸显示
  const sizeValueEl = this.contentEl.querySelector(".size-value");
  if (sizeValueEl) {
    sizeValueEl.textContent = `${newWidth} × ${newHeight} px`;
  }
  
  // 更新输入框的占位符
  const widthInput = this.contentEl.querySelector('input[placeholder^="当前:"]');
  const heightInput = this.contentEl.querySelectorAll('input[placeholder^="当前:"]')[1];
  
  if (widthInput) {
    widthInput.placeholder = `当前: ${newWidth}`;
  }
  if (heightInput) {
    heightInput.placeholder = `当前: ${newHeight}`;
  }
}

// 修改原有方法：支持部分尺寸应用
async applySingleCardSize(width, height) {
  try {
    if (!this.validateDimension(width) || !this.validateDimension(height)) {
      new Notice("尺寸值无效，请输入50-2000之间的数值");
      return;
    }
    
    const targetCard = this.cards[0];
    await this.cardService.unifyCardSizes([targetCard], { width, height });
    
    new Notice(`卡片尺寸已更新为 ${width}×${height}px`);
    this.close();
    
  } catch (error) {
    console.error("更新卡片尺寸失败:", error);
    new Notice("更新尺寸失败: " + error.message);
  }
}

// 新增方法：单个维度验证
validateDimension(value) {
  return !isNaN(value) && value >= 50 && value <= 2000;
}

## 主要改进内容

**操作列简化处理**：原有的卡片列表中"应用此尺寸"按钮已被移除，操作列现在显示简洁的占位符"-"，消除了单卡片场景下的无效操作选项。这一改动确保界面保持整洁并避免用户困惑。

**常用尺寸功能移除**：预设的常用尺寸选项已完全删除，简化了界面结构。用户现在可以直接进入自定义尺寸编辑区域，减少了不必要的中间步骤，提高了操作效率。

**独立维度控制机制**：自定义尺寸编辑功能已经过重新设计，支持宽度和高度的独立调整。每个维度都配备了专用的输入框、应用按钮和重置选项，允许用户精确控制所需修改的尺寸属性。

## 核心功能特性

**分离式尺寸调整**：新的界面设计提供了独立的宽度和高度编辑区域。当用户仅需调整一个维度时，可以只填写相应的输入框并应用，系统将自动保持另一维度的原有数值不变。

**智能输入处理**：系统现在支持部分输入的处理逻辑。如果某个维度的输入框保持空白，应用操作将使用该维度的当前值，确保用户可以灵活地进行单一维度或双维度的调整。

**实时反馈机制**：界面增加了动态的占位符显示和即时验证功能。输入框会显示当前尺寸值作为参考，按钮状态会根据输入有效性实时更新，为用户提供清晰的操作指导。
